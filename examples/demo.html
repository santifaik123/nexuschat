NexusChat

Premium embeddable AI chatbot for customer support. Click the button in the bottom-right corner to try it!

🚀 Quick Start

Embed NexusChat on any website with just two lines of code:

<!-- NexusChat Widget -->
<script>
  window.NexusChatConfig = {
    tenantId: 'nuvik',
    serverUrl: 'http://localhost:3000',
    // Optional customization:
    // name: 'My Support Bot',
    // primaryColor: '#4F46E5',
    // theme: 'dark',
    // language: 'es',
  };
</script>
<script src="http://localhost:3000/widget/nexuschat.min.js"></script>

🎛️ Configuration Options

{
  tenantId: 'nuvik',          // Your tenant ID
  serverUrl: 'http://...',      // Backend API URL
  name: 'NexusChat',            // Bot name
  primaryColor: '#4F46E5',      // Main color
  secondaryColor: '#7C3AED',    // Gradient color
  theme: 'light',               // 'light' or 'dark'
  position: 'bottom-right',     // 'bottom-right' or 'bottom-left'
  welcomeMessage: 'Hi! ...',    // Welcome text
  language: 'en',               // 'en', 'es', 'pt'
  quickReplies: ['Pricing'...], // Quick reply buttons
  borderRadius: 16,             // Window border radius
  showBranding: true,           // Show "Powered by NexusChat"
  logoUrl: '',                  // Custom logo URL
}
